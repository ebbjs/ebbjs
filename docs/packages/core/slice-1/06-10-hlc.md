### Task 7: Implement HLC pack/unpack/parse/format utilities

**Files:** `packages/core/src/hlc/pack.ts` (create)

**Depends on:** Task 6

```typescript
import { HLCComponents } from "./types.js";

export const COUNTER_BITS = 16n;
export const COUNTER_MASK = 0xFFFFn;

// Pack logical time (upper 48 bits) and counter (lower 16 bits) into a 64-bit integer
// Format: (logical_time << 16) | counter
export function pack(logicalTime: bigint, counter: bigint): bigint {
  return (logicalTime << COUNTER_BITS) | (counter & COUNTER_MASK);
}

// Unpack a 64-bit integer into HLC components
export function unpack(hlc: bigint): HLCComponents {
  return {
    logicalTime: hlc >> COUNTER_BITS,
    counter: hlc & COUNTER_MASK
  };
}

// Parse HLCTimestamp string (decimal) to bigint
export function parse(hlc: string): bigint {
  return BigInt(hlc);
}

// Format bigint to HLCTimestamp string (decimal)
export function format(hlc: bigint): string {
  return hlc.toString();
}
```

### Task 8: Implement HLC clock operations

**Files:** `packages/core/src/hlc/clock.ts` (create)

**Depends on:** Task 7

```typescript
import { HLCState, HLCComponents } from "./types.js";
import { COUNTER_BITS, COUNTER_MASK, DEFAULT_MAX_DRIFT_MS, MAX_FUTURE_DRIFT_MS, MAX_PAST_DRIFT_MS } from "./constants.js";
import { pack, unpack, parse, format } from "./pack.js";

// Create a new clock instance with optional max drift
export function createClock(maxDrift?: bigint): HLCState {
  return {
    l: 0n,
    c: 0n,
    maxDrift: maxDrift ?? DEFAULT_MAX_DRIFT_MS
  };
}

// Generate HLC for a local event
export function localEvent(state: HLCState): string {
  const now = BigInt(Date.now());
  
  if (now > state.l) {
    state.l = now;
    state.c = 0n;
  } else {
    state.c = (state.c + 1n) & COUNTER_MASK;  // Wrap at 65535
  }
  
  return format(pack(state.l, state.c));
}

// Receive a remote HLC, validate drift, advance local clock
export function receiveRemoteHLC(state: HLCState, remoteHlc: string): string {
  const remote = unpack(parse(remoteHlc));
  const now = BigInt(Date.now());
  
  if (remote.logicalTime > state.l) {
    state.l = remote.logicalTime;
    state.c = 0n;
  } else if (remote.logicalTime === state.l && remote.counter > state.c) {
    state.c = remote.counter;
  } else {
    state.c = (state.c + 1n) & COUNTER_MASK;
  }
  
  // Validate drift bounds (for server compatibility)
  const drift = now - state.l;
  if (drift > MAX_FUTURE_DRIFT_MS || -drift > MAX_PAST_DRIFT_MS) {
    throw new Error(`HLC drift exceeds bounds: ${drift}`);
  }
  
  return format(pack(state.l, state.c));
}
```

### Task 9: Implement HLC comparison and validation

**`src/hlc/compare.ts`:**

```typescript
import { parse } from "./pack.js";

// Compare two HLC timestamps
export function compare(a: string, b: string): -1 | 0 | 1 {
  const aBig = parse(a);
  const bBig = parse(b);
  
  if (aBig < bBig) return -1;
  if (aBig > bBig) return 1;
  return 0;
}

export function isBefore(a: string, b: string): boolean {
  return compare(a, b) === -1;
}

export function isAfter(a: string, b: string): boolean {
  return compare(a, b) === 1;
}
```

**`src/hlc/validate.ts`:**

```typescript
// Validate HLC format: must be a positive decimal string representing a positive integer
export function isValidHLC(hlc: string): boolean {
  if (!hlc || typeof hlc !== "string") return false;
  if (hlc.length === 0) return false;
  
  // Must be all digits
  if (!/^\d+$/.test(hlc)) return false;
  
  // Must be positive
  const n = BigInt(hlc);
  return n > 0n;
}
```
